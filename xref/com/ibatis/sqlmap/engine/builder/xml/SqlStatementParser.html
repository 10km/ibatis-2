26  import java.util.Properties;

28  public class SqlStatementParser {
29  
30    private XmlParserState state;
31  
32    public SqlStatementParser(XmlParserState config) {
33      this.state = config;
34    }
35  
36    public void parseGeneralStatement(Node node, MappedStatement statement) {
37  
38      // get attributes
39      Properties attributes = NodeletUtils.parseAttributes(node, state.getGlobalProps());
40      String id = attributes.getProperty("id");
41      String parameterMapName = state.applyNamespace(attributes.getProperty("parameterMap"));
42      String parameterClassName = attributes.getProperty("parameterClass");
43      String resultMapName = attributes.getProperty("resultMap");
44      String resultClassName = attributes.getProperty("resultClass");
45      String cacheModelName = state.applyNamespace(attributes.getProperty("cacheModel"));
46      String xmlResultName = attributes.getProperty("xmlResultName");
47      String resultSetType = attributes.getProperty("resultSetType");
48      String fetchSize = attributes.getProperty("fetchSize");
49      String allowRemapping = attributes.getProperty("remapResults");
50      String timeout = attributes.getProperty("timeout");
51  
52      if (state.isUseStatementNamespaces()) {
53        id = state.applyNamespace(id);
54      }
55      String[] additionalResultMapNames = null;
56      if (resultMapName != null) {
57        additionalResultMapNames = state.getAllButFirstToken(resultMapName);
58        resultMapName = state.getFirstToken(resultMapName);
59        resultMapName = state.applyNamespace(resultMapName);
60        for (int i = 0; i < additionalResultMapNames.length; i++) {
61          additionalResultMapNames[i] = state.applyNamespace(additionalResultMapNames[i]);
62        }
63      }
64  
65      String[] additionalResultClassNames = null;
66      if (resultClassName != null) {
67        additionalResultClassNames = state.getAllButFirstToken(resultClassName);
68        resultClassName = state.getFirstToken(resultClassName);
69      }
70      Class[] additionalResultClasses = null;
71      if (additionalResultClassNames != null) {
72        additionalResultClasses = new Class[additionalResultClassNames.length];
73        for (int i = 0; i < additionalResultClassNames.length; i++) {
74          additionalResultClasses[i] = resolveClass(additionalResultClassNames[i]);
75        }
76      }
77  
78      state.getConfig().getErrorContext().setMoreInfo("Check the parameter class.");
79      Class parameterClass = resolveClass(parameterClassName);
80  
81      state.getConfig().getErrorContext().setMoreInfo("Check the result class.");
82      Class resultClass = resolveClass(resultClassName);
83  
84      Integer timeoutInt = timeout == null ? null : new Integer(timeout);
85      Integer fetchSizeInt = fetchSize == null ? null : new Integer(fetchSize);
86      boolean allowRemappingBool = "true".equals(allowRemapping);
87  
88      MappedStatementConfig statementConf = state.getConfig().newMappedStatementConfig(id, statement,
89          new XMLSqlSource(state, node), parameterMapName, parameterClass, resultMapName, additionalResultMapNames,
90          resultClass, additionalResultClasses, resultSetType, fetchSizeInt, allowRemappingBool, timeoutInt, cacheModelName,
91          xmlResultName);
92  
93      findAndParseSelectKey(node, statementConf);
94    }
95  
96    private Class resolveClass(String resultClassName) {
97      try {
98        if (resultClassName != null) {
99          return Resources.classForName(state.getConfig().getTypeHandlerFactory().resolveAlias(resultClassName));
100       } else {
101         return null;
102       }
103     } catch (ClassNotFoundException e) {
104       throw new SqlMapException("Error.  Could not initialize class.  Cause: " + e, e);
105     }
106   }
107 
108   private void findAndParseSelectKey(Node node, MappedStatementConfig config) {
109       state.getConfig().getErrorContext().setActivity("parsing select key tags");
110       boolean foundSQLFirst = false;
111       NodeList children = node.getChildNodes();
112       for (int i = 0; i < children.getLength(); i++) {
113         Node child = children.item(i);
114         if (child.getNodeType() == Node.CDATA_SECTION_NODE
115             || child.getNodeType() == Node.TEXT_NODE) {
116           String data = ((CharacterData) child).getData();
117           if (data.trim().length() > 0) {
118             foundSQLFirst = true;
119           }
120         } else if (child.getNodeType() == Node.ELEMENT_NODE
121             && "selectKey".equals(child.getNodeName())) {
122           Properties attributes = NodeletUtils.parseAttributes(child, state.getGlobalProps());
123           String keyPropName = attributes.getProperty("keyProperty");
124           String resultClassName = attributes.getProperty("resultClass");
125           String type = attributes.getProperty("type");
126           config.setSelectKeyStatement(new XMLSqlSource(state, child), resultClassName, keyPropName, foundSQLFirst, type);
127           break;
128         }
129       }
130       state.getConfig().getErrorContext().setMoreInfo(null);
131     
132   }
