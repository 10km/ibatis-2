130   public void newTypeHandler(Class javaType, String jdbcType, Object callback) {
131     try {
132       errorContext.setActivity("building a building custom type handler");
133       TypeHandlerFactory typeHandlerFactory = client.getDelegate().getTypeHandlerFactory();
134       TypeHandler typeHandler;
135       if (callback instanceof TypeHandlerCallback) {
136         typeHandler = new CustomTypeHandler((TypeHandlerCallback) callback);
137       } else if (callback instanceof TypeHandler) {
138         typeHandler = (TypeHandler) callback;
139       } else {
140         throw new RuntimeException("The object '" + callback
141             + "' is not a valid implementation of TypeHandler or TypeHandlerCallback");
142       }
143       errorContext.setMoreInfo("Check the javaType attribute '" + javaType
144           + "' (must be a classname) or the jdbcType '" + jdbcType + "' (must be a JDBC type name).");
145       if (jdbcType != null && jdbcType.length() > 0) {
146         typeHandlerFactory.register(javaType, jdbcType, typeHandler);
147       } else {
148         typeHandlerFactory.register(javaType, typeHandler);
149       }
150     } catch (Exception e) {
151       throw new SqlMapException("Error registering occurred.  Cause: " + e, e);
152     }
153     errorContext.setMoreInfo(null);
154     errorContext.setObjectId(null);
155   }

189   TypeHandler resolveTypeHandler(TypeHandlerFactory typeHandlerFactory, Class clazz, String propertyName,
190       Class javaType, String jdbcType, boolean useSetterToResolve) {
191     TypeHandler handler;
192     if (clazz == null) {
193       // Unknown
194       handler = typeHandlerFactory.getUnkownTypeHandler();
195     } else if (DomTypeMarker.class.isAssignableFrom(clazz)) {
196       // DOM
197       handler = typeHandlerFactory.getTypeHandler(String.class, jdbcType);
198     } else if (java.util.Map.class.isAssignableFrom(clazz)) {
199       // Map
200       if (javaType == null) {
201         handler = typeHandlerFactory.getUnkownTypeHandler(); // BUG 1012591 -
202                                                              // typeHandlerFactory.getTypeHandler(java.lang.Object.class,
203                                                              // jdbcType);
204       } else {
205         handler = typeHandlerFactory.getTypeHandler(javaType, jdbcType);
206       }
207     } else if (typeHandlerFactory.getTypeHandler(clazz, jdbcType) != null) {
208       // Primitive
209       handler = typeHandlerFactory.getTypeHandler(clazz, jdbcType);
210     } else {
211       // JavaBean
212       if (javaType == null) {
213         if (useSetterToResolve) {
214           Class type = PROBE.getPropertyTypeForSetter(clazz, propertyName);
215           handler = typeHandlerFactory.getTypeHandler(type, jdbcType);
216         } else {
217           Class type = PROBE.getPropertyTypeForGetter(clazz, propertyName);
218           handler = typeHandlerFactory.getTypeHandler(type, jdbcType);
219         }
220       } else {
221         handler = typeHandlerFactory.getTypeHandler(javaType, jdbcType);
222       }
223     }
224     return handler;
225   }

227   private void registerDefaultTypeAliases() {
228     // TRANSACTION ALIASES
229     typeHandlerFactory.putTypeAlias("JDBC", JdbcTransactionConfig.class.getName());
230     typeHandlerFactory.putTypeAlias("JTA", JtaTransactionConfig.class.getName());
231     typeHandlerFactory.putTypeAlias("EXTERNAL", ExternalTransactionConfig.class.getName());
232 
233     // DATA SOURCE ALIASES
234     typeHandlerFactory.putTypeAlias("SIMPLE", SimpleDataSourceFactory.class.getName());
235     typeHandlerFactory.putTypeAlias("DBCP", DbcpDataSourceFactory.class.getName());
236     typeHandlerFactory.putTypeAlias("JNDI", JndiDataSourceFactory.class.getName());
237 
238     // CACHE ALIASES
239     typeHandlerFactory.putTypeAlias("FIFO", FifoCacheController.class.getName());
240     typeHandlerFactory.putTypeAlias("LRU", LruCacheController.class.getName());
241     typeHandlerFactory.putTypeAlias("MEMORY", MemoryCacheController.class.getName());
242     // use a string for OSCache to avoid unnecessary loading of properties upon init
243     typeHandlerFactory.putTypeAlias("OSCACHE", "com.ibatis.sqlmap.engine.cache.oscache.OSCacheController");
244 
245     // TYPE ALIASEs
246     typeHandlerFactory.putTypeAlias("dom", DomTypeMarker.class.getName());
247     typeHandlerFactory.putTypeAlias("domCollection", DomCollectionTypeMarker.class.getName());
248     typeHandlerFactory.putTypeAlias("xml", XmlTypeMarker.class.getName());
249     typeHandlerFactory.putTypeAlias("xmlCollection", XmlCollectionTypeMarker.class.getName());
250   }

252   private void wireUpCacheModels() {
253     // Wire Up Cache Models
254     Iterator cacheNames = client.getDelegate().getCacheModelNames();
255     while (cacheNames.hasNext()) {
256       String cacheName = (String) cacheNames.next();
257       CacheModel cacheModel = client.getDelegate().getCacheModel(cacheName);
258       Iterator statementNames = cacheModel.getFlushTriggerStatementNames();
259       while (statementNames.hasNext()) {
260         String statementName = (String) statementNames.next();
261         MappedStatement statement = client.getDelegate().getMappedStatement(statementName);
262         if (statement != null) {
263           statement.addExecuteListener(cacheModel);
264         } else {
265           throw new RuntimeException("Could not find statement named '" + statementName
266               + "' for use as a flush trigger for the cache model named '" + cacheName + "'.");
267         }
268       }
269     }
270   }

272   private void bindResultMapDiscriminators() {
273     // Bind discriminators
274     Iterator names = delegate.getResultMapNames();
275     while (names.hasNext()) {
276       String name = (String) names.next();
277       ResultMap rm = delegate.getResultMap(name);
278       Discriminator disc = rm.getDiscriminator();
279       if (disc != null) {
280         disc.bindSubMaps();
281       }
282     }
