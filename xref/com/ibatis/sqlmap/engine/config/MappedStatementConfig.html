36  public class MappedStatementConfig {
37    private static final Probe PROBE = ProbeFactory.getProbe();
38    private static final InlineParameterMapParser PARAM_PARSER = new InlineParameterMapParser();
39    private ErrorContext errorContext;
40    private SqlMapClientImpl client;
41    private TypeHandlerFactory typeHandlerFactory;
42    private MappedStatement mappedStatement;
43    private MappedStatement rootStatement;

45    MappedStatementConfig(SqlMapConfiguration config, String id, MappedStatement statement, SqlSource processor,
46                          String parameterMapName, Class parameterClass, String resultMapName,
47                          String[] additionalResultMapNames, Class resultClass, Class[] additionalResultClasses,
48                          String cacheModelName, String resultSetType, Integer fetchSize, boolean allowRemapping,
49                          Integer timeout, Integer defaultStatementTimeout, String xmlResultName) {
50      this.errorContext = config.getErrorContext();
51      this.client = config.getClient();
52      SqlMapExecutorDelegate delegate = client.getDelegate();
53      this.typeHandlerFactory = config.getTypeHandlerFactory();
54      errorContext.setActivity("parsing a mapped statement");
55      errorContext.setObjectId(id + " statement");
56      errorContext.setMoreInfo("Check the result map name.");
57      if (resultMapName != null) {
58        statement.setResultMap(client.getDelegate().getResultMap(resultMapName));
59        if (additionalResultMapNames != null) {
60          for (int i = 0; i < additionalResultMapNames.length; i++) {
61            statement.addResultMap(client.getDelegate().getResultMap(additionalResultMapNames[i]));
62          }
63        }
64      }
65      errorContext.setMoreInfo("Check the parameter map name.");
66      if (parameterMapName != null) {
67        statement.setParameterMap(client.getDelegate().getParameterMap(parameterMapName));
68      }
69      statement.setId(id);
70      statement.setResource(errorContext.getResource());
71      if (resultSetType != null) {
72        if ("FORWARD_ONLY".equals(resultSetType)) {
73          statement.setResultSetType(new Integer(ResultSet.TYPE_FORWARD_ONLY));
74        } else if ("SCROLL_INSENSITIVE".equals(resultSetType)) {
75          statement.setResultSetType(new Integer(ResultSet.TYPE_SCROLL_INSENSITIVE));
76        } else if ("SCROLL_SENSITIVE".equals(resultSetType)) {
77          statement.setResultSetType(new Integer(ResultSet.TYPE_SCROLL_SENSITIVE));
78        }
79      }
80      if (fetchSize != null) {
81        statement.setFetchSize(fetchSize);
82      }
83  
84      // set parameter class either from attribute or from map (make sure to match)
85      ParameterMap parameterMap = statement.getParameterMap();
86      if (parameterMap == null) {
87        statement.setParameterClass(parameterClass);
88      } else {
89        statement.setParameterClass(parameterMap.getParameterClass());
90      }
91  
92      // process SQL statement, including inline parameter maps
93      errorContext.setMoreInfo("Check the SQL statement.");
94      Sql sql = processor.getSql();
95      setSqlForStatement(statement, sql);
96  
97      // set up either null result map or automatic result mapping
98      ResultMap resultMap = (ResultMap) statement.getResultMap();
99      if (resultMap == null && resultClass == null) {
100       statement.setResultMap(null);
101     } else if (resultMap == null) {
102       resultMap = buildAutoResultMap(allowRemapping, statement, resultClass, xmlResultName);
103       statement.setResultMap(resultMap);
104       if (additionalResultClasses != null) {
105         for (int i = 0; i < additionalResultClasses.length; i++) {
106           statement.addResultMap(buildAutoResultMap(allowRemapping, statement, additionalResultClasses[i], xmlResultName));
107         }
108       }
109 
110     }
111     statement.setTimeout(defaultStatementTimeout);
112     if (timeout != null) {
113       try {
114         statement.setTimeout(timeout);
115       } catch (NumberFormatException e) {
116         throw new SqlMapException("Specified timeout value for statement " + statement.getId() + " is not a valid integer");
117       }
118     }
119     errorContext.setMoreInfo(null);
120     errorContext.setObjectId(null);
121     statement.setSqlMapClient(client);
122     if (cacheModelName != null && cacheModelName.length() > 0 && client.getDelegate().isCacheModelsEnabled()) {
123       CacheModel cacheModel = client.getDelegate().getCacheModel(cacheModelName);
124       mappedStatement = new CachingStatement(statement, cacheModel);
125     } else {
126       mappedStatement = statement;
127     }
128     rootStatement = statement;
129     delegate.addMappedStatement(mappedStatement);
130   }

132   public void setSelectKeyStatement(SqlSource processor, String resultClassName, String keyPropName, boolean runAfterSQL, String type) {
133     if (rootStatement instanceof InsertStatement) {
134       InsertStatement insertStatement = ((InsertStatement) rootStatement);
135       Class parameterClass = insertStatement.getParameterClass();
136       errorContext.setActivity("parsing a select key");
137       SelectKeyStatement selectKeyStatement = new SelectKeyStatement();
138       resultClassName = typeHandlerFactory.resolveAlias(resultClassName);
139       Class resultClass = null;
140 
141       // get parameter and result maps
142       selectKeyStatement.setSqlMapClient(client);
143       selectKeyStatement.setId(insertStatement.getId() + "-SelectKey");
144       selectKeyStatement.setResource(errorContext.getResource());
145       selectKeyStatement.setKeyProperty(keyPropName);
146       selectKeyStatement.setRunAfterSQL(runAfterSQL);
147       // process the type (pre or post) attribute
148       if (type != null) {
149         selectKeyStatement.setRunAfterSQL("post".equals(type));
150       }
151       try {
152         if (resultClassName != null) {
153           errorContext.setMoreInfo("Check the select key result class.");
154           resultClass = Resources.classForName(resultClassName);
155         } else {
156           if (keyPropName != null && parameterClass != null) {
157             resultClass = PROBE.getPropertyTypeForSetter(parameterClass, selectKeyStatement.getKeyProperty());
158           }
159         }
160       } catch (ClassNotFoundException e) {
161         throw new SqlMapException("Error.  Could not set result class.  Cause: " + e, e);
162       }
163       if (resultClass == null) {
164         resultClass = Object.class;
165       }
166 
167       // process SQL statement, including inline parameter maps
168       errorContext.setMoreInfo("Check the select key SQL statement.");
169       Sql sql = processor.getSql();
170       setSqlForStatement(selectKeyStatement, sql);
171       ResultMap resultMap;
172       resultMap = new AutoResultMap(client.getDelegate(), false);
173       resultMap.setId(selectKeyStatement.getId() + "-AutoResultMap");
174       resultMap.setResultClass(resultClass);
175       resultMap.setResource(selectKeyStatement.getResource());
176       selectKeyStatement.setResultMap(resultMap);
177       errorContext.setMoreInfo(null);
178       insertStatement.setSelectKeyStatement(selectKeyStatement);
179     } else {
180       throw new SqlMapException("You cant set a select key statement on statement named " + rootStatement.getId() + " because it is not an InsertStatement.");
181     }
182   }

184   private void setSqlForStatement(MappedStatement statement, Sql sql) {
185     if (sql instanceof DynamicSql) {
186       statement.setSql(sql);
187     } else {
188       applyInlineParameterMap(statement, sql.getSql(null, null));
189     }
190   }
191 
192   private void applyInlineParameterMap(MappedStatement statement, String sqlStatement) {
193     String newSql = sqlStatement;
194     errorContext.setActivity("building an inline parameter map");
195     ParameterMap parameterMap = statement.getParameterMap();
196     errorContext.setMoreInfo("Check the inline parameters.");
197     if (parameterMap == null) {
198       ParameterMap map;
199       map = new ParameterMap(client.getDelegate());
200       map.setId(statement.getId() + "-InlineParameterMap");
201       map.setParameterClass(statement.getParameterClass());
202       map.setResource(statement.getResource());
203       statement.setParameterMap(map);
204       SqlText sqlText = PARAM_PARSER.parseInlineParameterMap(client.getDelegate().getTypeHandlerFactory(), newSql, statement.getParameterClass());
205       newSql = sqlText.getText();
206       List mappingList = Arrays.asList(sqlText.getParameterMappings());
207       map.setParameterMappingList(mappingList);
208     }
209     Sql sql;
210     if (SimpleDynamicSql.isSimpleDynamicSql(newSql)) {
211       sql = new SimpleDynamicSql(client.getDelegate(), newSql);
212     } else {
213       sql = new StaticSql(newSql);
214     }
215     statement.setSql(sql);
216 
217   }

219   private ResultMap buildAutoResultMap(boolean allowRemapping, MappedStatement statement, Class firstResultClass, String xmlResultName) {
220     ResultMap resultMap;
221     resultMap = new AutoResultMap(client.getDelegate(), allowRemapping);
222     resultMap.setId(statement.getId() + "-AutoResultMap");
223     resultMap.setResultClass(firstResultClass);
224     resultMap.setXmlName(xmlResultName);
225     resultMap.setResource(statement.getResource());
226     return resultMap;
227   }
