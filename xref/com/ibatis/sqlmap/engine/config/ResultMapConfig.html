26  public class ResultMapConfig {
27    private SqlMapConfiguration config;
28    private ErrorContext errorContext;
29    private SqlMapClientImpl client;
30    private SqlMapExecutorDelegate delegate;
31    private TypeHandlerFactory typeHandlerFactory;
32    private ResultMap resultMap;
33    private List resultMappingList;
34    private int resultMappingIndex;
35    private Discriminator discriminator;
36  
37    ResultMapConfig(SqlMapConfiguration config, String id, Class resultClass, String groupBy, String extendsResultMap,
38        String xmlName) {
39      this.config = config;
40      this.errorContext = config.getErrorContext();
41      this.client = config.getClient();
42      this.delegate = config.getDelegate();
43      this.typeHandlerFactory = config.getTypeHandlerFactory();
44      this.resultMap = new ResultMap(client.getDelegate());
45      this.resultMappingList = new ArrayList();
46      errorContext.setActivity("building a result map");
47      errorContext.setObjectId(id + " result map");
48      resultMap.setId(id);
49      resultMap.setXmlName(xmlName);
50      resultMap.setResource(errorContext.getResource());
51      if (groupBy != null && groupBy.length() > 0) {
52        StringTokenizer parser = new StringTokenizer(groupBy, ", ", false);
53        while (parser.hasMoreTokens()) {
54          resultMap.addGroupByProperty(parser.nextToken());
55        }
56      }
57      resultMap.setResultClass(resultClass);
58      errorContext.setMoreInfo("Check the extended result map.");
59      if (extendsResultMap != null) {
60        ResultMap extendedResultMap = (ResultMap) client.getDelegate().getResultMap(extendsResultMap);
61        ResultMapping[] resultMappings = extendedResultMap.getResultMappings();
62        for (int i = 0; i < resultMappings.length; i++) {
63          resultMappingList.add(resultMappings[i]);
64        }
65        List nestedResultMappings = extendedResultMap.getNestedResultMappings();
66        if (nestedResultMappings != null) {
67          Iterator iter = nestedResultMappings.iterator();
68          while (iter.hasNext()) {
69            resultMap.addNestedResultMappings((ResultMapping) iter.next());
70          }
71        }
72        if (groupBy == null || groupBy.length() == 0) {
73          if (extendedResultMap.hasGroupBy()) {
74            Iterator i = extendedResultMap.groupByProps();
75            while (i.hasNext()) {
76              resultMap.addGroupByProperty((String) i.next());
77            }
78          }
79        }
80      }
81      errorContext.setMoreInfo("Check the result mappings.");
82      resultMappingIndex = resultMappingList.size();
83      resultMap.setResultMappingList(resultMappingList);
84      client.getDelegate().addResultMap(resultMap);
85    }
86  
87    public void setDiscriminator(String columnName, Integer columnIndex, Class javaClass, String jdbcType,
88        String nullValue, Object typeHandlerImpl) {
89      TypeHandler handler;
90      if (typeHandlerImpl != null) {
91        if (typeHandlerImpl instanceof TypeHandlerCallback) {
92          handler = new CustomTypeHandler((TypeHandlerCallback) typeHandlerImpl);
93        } else if (typeHandlerImpl instanceof TypeHandler) {
94          handler = (TypeHandler) typeHandlerImpl;
95        } else {
96          throw new RuntimeException("The class '' is not a valid implementation of TypeHandler or TypeHandlerCallback");
97        }
98      } else {
99        handler = config.resolveTypeHandler(client.getDelegate().getTypeHandlerFactory(), resultMap.getResultClass(), "",
100           javaClass, jdbcType, true);
101     }
102     ResultMapping mapping = new ResultMapping();
103     mapping.setColumnName(columnName);
104     mapping.setJdbcTypeName(jdbcType);
105     mapping.setTypeHandler(handler);
106     mapping.setNullValue(nullValue);
107     mapping.setJavaType(javaClass);
108     if (columnIndex != null) {
109       mapping.setColumnIndex(columnIndex.intValue());
110     }
111     discriminator = new Discriminator(delegate, mapping);
112     resultMap.setDiscriminator(discriminator);
113   }
114 
115   public void addDiscriminatorSubMap(Object value, String resultMap) {
116     if (discriminator == null) {
117       throw new RuntimeException("The discriminator is null, but somehow a subMap was reached.  This is a bug.");
118     }
119     discriminator.addSubMap(value.toString(), resultMap);
120   }
121 
122   public void addResultMapping(String propertyName, String columnName, Integer columnIndex, Class javaClass,
123       String jdbcType, String nullValue, String notNullColumn, String statementName, String resultMapName, Object impl) {
124     errorContext.setObjectId(propertyName + " mapping of the " + resultMap.getId() + " result map");
125     TypeHandler handler;
126     if (impl != null) {
127       if (impl instanceof TypeHandlerCallback) {
128         handler = new CustomTypeHandler((TypeHandlerCallback) impl);
129       } else if (impl instanceof TypeHandler) {
130         handler = (TypeHandler) impl;
131       } else {
132         throw new RuntimeException("The class '" + impl
133             + "' is not a valid implementation of TypeHandler or TypeHandlerCallback");
134       }
135     } else {
136       handler = config.resolveTypeHandler(client.getDelegate().getTypeHandlerFactory(), resultMap.getResultClass(),
137           propertyName, javaClass, jdbcType, true);
138     }
139     ResultMapping mapping = new ResultMapping();
140     mapping.setPropertyName(propertyName);
141     mapping.setColumnName(columnName);
142     mapping.setJdbcTypeName(jdbcType);
143     mapping.setTypeHandler(handler);
144     mapping.setNullValue(nullValue);
145     mapping.setNotNullColumn(notNullColumn);
146     mapping.setStatementName(statementName);
147     mapping.setNestedResultMapName(resultMapName);
148     if (resultMapName != null && resultMapName.length() > 0) {
149       resultMap.addNestedResultMappings(mapping);
150     }
151     mapping.setJavaType(javaClass);
152     if (columnIndex != null) {
153       mapping.setColumnIndex(columnIndex.intValue());
154     } else {
155       resultMappingIndex++;
156       mapping.setColumnIndex(resultMappingIndex);
157     }
158     resultMappingList.add(mapping);
159     resultMap.setResultMappingList(resultMappingList);
160   }
