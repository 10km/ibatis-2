61    public int executeUpdate(StatementScope statementScope, Transaction trans, Object parameterObject)
62        throws SQLException {
63      ErrorContext errorContext = statementScope.getErrorContext();
64      errorContext.setActivity("preparing the mapped statement for execution");
65      errorContext.setObjectId(this.getId());
66      errorContext.setResource(this.getResource());
67  
68      statementScope.getSession().setCommitRequired(true);
69  
70      try {
71        parameterObject = validateParameter(parameterObject);
72  
73        Sql sql = getSql();
74  
75        errorContext.setMoreInfo("Check the parameter map.");
76        ParameterMap parameterMap = sql.getParameterMap(statementScope, parameterObject);
77  
78        errorContext.setMoreInfo("Check the result map.");
79        ResultMap resultMap = sql.getResultMap(statementScope, parameterObject);
80  
81        statementScope.setResultMap(resultMap);
82        statementScope.setParameterMap(parameterMap);
83  
84        int rows = 0;
85  
86        errorContext.setMoreInfo("Check the parameter map.");
87        Object[] parameters = parameterMap.getParameterObjectValues(statementScope, parameterObject);
88  
89        errorContext.setMoreInfo("Check the SQL statement.");
90        String sqlString = sql.getSql(statementScope, parameterObject);
91  
92        errorContext.setActivity("executing mapped statement");
93        errorContext.setMoreInfo("Check the statement or the result map.");
94        rows = sqlExecuteUpdate(statementScope, trans.getConnection(), sqlString, parameters);
95  
96        errorContext.setMoreInfo("Check the output parameters.");
97        if (parameterObject != null) {
98          postProcessParameterObject(statementScope, parameterObject, parameters);
99        }
100 
101       errorContext.reset();
102       sql.cleanup(statementScope);
103       notifyListeners();
104       return rows;
105     } catch (SQLException e) {
106       errorContext.setCause(e);
107       throw new NestedSQLException(errorContext.toString(), e.getSQLState(), e.getErrorCode(), e);
108     } catch (Exception e) {
109       errorContext.setCause(e);
110       throw new NestedSQLException(errorContext.toString(), e);
111     }
112   }

114   public Object executeQueryForObject(StatementScope statementScope, Transaction trans, Object parameterObject,
115       Object resultObject) throws SQLException {
116     try {
117       Object object = null;
118 
119       DefaultRowHandler rowHandler = new DefaultRowHandler();
120       executeQueryWithCallback(statementScope, trans.getConnection(), parameterObject, resultObject, rowHandler,
121           SqlExecutor.NO_SKIPPED_RESULTS, SqlExecutor.NO_MAXIMUM_RESULTS);
122       List list = rowHandler.getList();
123 
124       if (list.size() > 1) {
125         throw new SQLException("Error: executeQueryForObject returned too many results.");
126       } else if (list.size() > 0) {
127         object = list.get(0);
128       }
129 
130       return object;
131     } catch (TransactionException e) {
132       throw new NestedSQLException("Error getting Connection from Transaction.  Cause: " + e, e);
133     }
134   }

136   public List executeQueryForList(StatementScope statementScope, Transaction trans, Object parameterObject,
137       int skipResults, int maxResults) throws SQLException {
138     try {
139       DefaultRowHandler rowHandler = new DefaultRowHandler();

141           maxResults);
142       return rowHandler.getList();
143     } catch (TransactionException e) {
144       throw new NestedSQLException("Error getting Connection from Transaction.  Cause: " + e, e);
145     }

162   protected void executeQueryWithCallback(StatementScope statementScope, Connection conn, Object parameterObject,
163       Object resultObject, RowHandler rowHandler, int skipResults, int maxResults) throws SQLException {
164     ErrorContext errorContext = statementScope.getErrorContext();
165     errorContext.setActivity("preparing the mapped statement for execution");
166     errorContext.setObjectId(this.getId());
167     errorContext.setResource(this.getResource());
168 
169     try {
170       parameterObject = validateParameter(parameterObject);
171 
172       Sql sql = getSql();
173 
174       errorContext.setMoreInfo("Check the parameter map.");
175       ParameterMap parameterMap = sql.getParameterMap(statementScope, parameterObject);
176 
177       errorContext.setMoreInfo("Check the result map.");
178       ResultMap resultMap = sql.getResultMap(statementScope, parameterObject);
179 
180       statementScope.setResultMap(resultMap);
181       statementScope.setParameterMap(parameterMap);
182 
183       errorContext.setMoreInfo("Check the parameter map.");
184       Object[] parameters = parameterMap.getParameterObjectValues(statementScope, parameterObject);
185 
186       errorContext.setMoreInfo("Check the SQL statement.");
187       String sqlString = sql.getSql(statementScope, parameterObject);
188 
189       errorContext.setActivity("executing mapped statement");
190       errorContext.setMoreInfo("Check the SQL statement or the result map.");
191       RowHandlerCallback callback = new RowHandlerCallback(resultMap, resultObject, rowHandler);
192       sqlExecuteQuery(statementScope, conn, sqlString, parameters, skipResults, maxResults, callback);
193 
194       errorContext.setMoreInfo("Check the output parameters.");
195       if (parameterObject != null) {
196         postProcessParameterObject(statementScope, parameterObject, parameters);
197       }
198 
199       errorContext.reset();
200       sql.cleanup(statementScope);
201       notifyListeners();
202     } catch (SQLException e) {
203       errorContext.setCause(e);
204       throw new NestedSQLException(errorContext.toString(), e.getSQLState(), e.getErrorCode(), e);
205     } catch (Exception e) {
206       errorContext.setCause(e);
207       throw new NestedSQLException(errorContext.toString(), e);
208     }
209   }

214   protected int sqlExecuteUpdate(StatementScope statementScope, Connection conn, String sqlString, Object[] parameters)
215       throws SQLException {
216     if (statementScope.getSession().isInBatch()) {
217       getSqlExecutor().addBatch(statementScope, conn, sqlString, parameters);
218       return 0;
219     } else {
220       return getSqlExecutor().executeUpdate(statementScope, conn, sqlString, parameters);
221     }
222   }

224   protected void sqlExecuteQuery(StatementScope statementScope, Connection conn, String sqlString, Object[] parameters,
225       int skipResults, int maxResults, RowHandlerCallback callback) throws SQLException {
226     getSqlExecutor().executeQuery(statementScope, conn, sqlString, parameters, skipResults, maxResults, callback);
227   }

229   protected Object validateParameter(Object param) throws SQLException {
230     Object newParam = param;
231     Class parameterClass = getParameterClass();
232     if (newParam != null && parameterClass != null) {
233       if (DomTypeMarker.class.isAssignableFrom(parameterClass)) {
234         if (XmlTypeMarker.class.isAssignableFrom(parameterClass)) {
235           if (!(newParam instanceof String) && !(newParam instanceof Document)) {
236             throw new SQLException("Invalid parameter object type.  Expected '" + String.class.getName() + "' or '"
237                 + Document.class.getName() + "' but found '" + newParam.getClass().getName() + "'.");
238           }
239           if (!(newParam instanceof Document)) {
240             newParam = stringToDocument((String) newParam);
241           }
242         } else {
243           if (!Document.class.isAssignableFrom(newParam.getClass())) {
244             throw new SQLException("Invalid parameter object type.  Expected '" + Document.class.getName()
245                 + "' but found '" + newParam.getClass().getName() + "'.");
246           }
247         }
248       } else {
249         if (!parameterClass.isAssignableFrom(newParam.getClass())) {
250           throw new SQLException("Invalid parameter object type.  Expected '" + parameterClass.getName()
251               + "' but found '" + newParam.getClass().getName() + "'.");
252         }
253       }
254     }
255     return newParam;
256   }

258   private Document stringToDocument(String s) {
259     try {
260       DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
261       DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
262       return documentBuilder.parse(new ReaderInputStream(new StringReader(s)));
263     } catch (Exception e) {
264       throw new RuntimeException("Error occurred.  Cause: " + e, e);
265     }
266   }

332   public CacheKey getCacheKey(StatementScope statementScope, Object parameterObject) {
333     Sql sql = statementScope.getSql();
334     ParameterMap pmap = sql.getParameterMap(statementScope, parameterObject);
335     CacheKey cacheKey = pmap.getCacheKey(statementScope, parameterObject);
336     cacheKey.update(id);
337 
338     // I am not sure how any clustered cache solution would ever have had any cache hits against
339     // replicated objects. I could not make it happen
340     // The baseCacheKey value which was being used in the update below is consistent across
341     // JVMInstances on the same machine
342     // but it's not consistent across machines, and therefore breaks clustered caching.
343 
344     // What would happen is the cache values were being replicated across machines but there
345     // were never any cache hits for cached objects on
346     // anything but the original machine an object was created on.
347 
348     // After reviewing this implementation I could not figure out why baseCacheKey is used for
349     // this anyway as it's not needed, so I removed it.
350     // The values used from the pmap.getCacheKey, plus id, plus the params below are unique and
351     // the same accross machines, so now I get replicated
352     // cache hits when I force failover in my cluster
353 
354     // I wish I could make a unit test for this, but I can't do it as the old implementaion
355     // works on 1 machine, but fails across machines.
356     // cacheKey.update(baseCacheKey);
357 
358     cacheKey.update(sql.getSql(statementScope, parameterObject)); // Fixes bug 953001
359     return cacheKey;
360   }
