16  package com.ibatis.sqlmap.engine.execution;
17  
18  import java.sql.Connection;
19  import java.sql.SQLException;
20  import java.util.List;
21  import java.util.Properties;
22  
23  import com.ibatis.sqlmap.engine.config.SqlMapConfiguration;
24  import com.ibatis.sqlmap.engine.mapping.statement.RowHandlerCallback;
25  import com.ibatis.sqlmap.engine.scope.SessionScope;
26  import com.ibatis.sqlmap.engine.scope.StatementScope;
27  
28  /**
29   * Classes responsible for executing the SQL implement this interface
30   * Support for custom SQL Executors.
31   */
32  public interface SqlExecutor {
33    //
34    // Constants
35    //
36    /**
37     * Constant to let us know not to skip anything
38     */
39    int NO_SKIPPED_RESULTS = 0;
40    /**
41     * Constant to let us know to include all records
42     */
43    int NO_MAXIMUM_RESULTS = -999999;
44  
45    /**
46     * Execute an update
47     *
48     * @param statementScope    - the request scope
49     * @param conn       - the database connection
50     * @param sql        - the sql statement to execute
51     * @param parameters - the parameters for the sql statement
52     * @return - the number of records changed
53     * @throws SQLException - if the update fails
54     */
55    public int executeUpdate(StatementScope statementScope, Connection conn, String sql, Object[] parameters) throws SQLException;
56  	
57    /**
58     * Adds a statement to a batch
59     *
60     * @param statementScope    - the request scope
61     * @param conn       - the database connection
62     * @param sql        - the sql statement
63     * @param parameters - the parameters for the statement
64     * @throws SQLException - if the statement fails
65     */
66    public void addBatch(StatementScope statementScope, Connection conn, String sql, Object[] parameters) throws SQLException;
67      
68    /**
69     * Execute a batch of statements
70     *
71     * @param sessionScope - the session scope
72     * @return - the number of rows impacted by the batch
73     * @throws SQLException - if a statement fails
74     */
75    public int executeBatch(SessionScope sessionScope) throws SQLException;
76    
77    /**
78     * Execute a batch of statements
79     *
80     * @param sessionScope - the session scope
81     * @return - a List of BatchResult objects (may be null if no batch
82     *         has been initiated).  There will be one BatchResult object in the
83     *         list for each sub-batch executed
84     * @throws SQLException   if a database access error occurs, or the drive
85     *                        does not support batch statements
86     * @throws BatchException if the driver throws BatchUpdateException
87     */
88    public List executeBatchDetailed(SessionScope sessionScope) throws SQLException, BatchException;
89    
90    /**
91     * Long form of the method to execute a query
92     *
93     * @param statementScope     - the request scope
94     * @param conn        - the database connection
95     * @param sql         - the SQL statement to execute
96     * @param parameters  - the parameters for the statement
97     * @param skipResults - the number of results to skip
98     * @param maxResults  - the maximum number of results to return
99     * @param callback    - the row handler for the query
100    * @throws SQLException - if the query fails
101    */
102   public void executeQuery(StatementScope statementScope, Connection conn, String sql, Object[] parameters, int skipResults, int maxResults, RowHandlerCallback callback) throws SQLException;
103   
104   /**
105    * Execute a stored procedure that updates data
106    *
107    * @param statementScope    - the request scope
108    * @param conn       - the database connection
109    * @param sql        - the SQL to call the procedure
110    * @param parameters - the parameters for the procedure
111    * @return - the rows impacted by the procedure
112    * @throws SQLException - if the procedure fails
113    */
114   public int executeUpdateProcedure(StatementScope statementScope, Connection conn, String sql, Object[] parameters) throws SQLException;
115   
116   /**
117    * Execute a stored procedure
118    *
119    * @param statementScope     - the request scope
120    * @param conn        - the database connection
121    * @param sql         - the sql to call the procedure
122    * @param parameters  - the parameters for the procedure
123    * @param skipResults - the number of results to skip
124    * @param maxResults  - the maximum number of results to return
125    * @param callback    - a row handler for processing the results
126    * @throws SQLException - if the procedure fails
127    */
128   public void executeQueryProcedure(StatementScope statementScope, Connection conn, String sql, Object[] parameters, int skipResults, int maxResults, RowHandlerCallback callback) throws SQLException;
129   
130   /**
131    * Clean up any batches on the session
132    *
133    * @param sessionScope - the session to clean up
134    */
135   public void cleanup(SessionScope sessionScope);
136   
137   /**
138    * Initializing SQL Executor by passing configuration and global properties.
139    * 
140    * @param config - the configuration class, which contains complete configuration info
141    * @param globalProps
142    */
143   public void init(SqlMapConfiguration config, Properties globalProps);
144 }
