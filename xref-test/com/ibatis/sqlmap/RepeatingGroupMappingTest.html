41    public void testNestedProperties() throws Exception {
42      List list = sqlMap.queryForList("getFish",null);
43      assertEquals (1, list.size());
44  
45      Category cat = (Category)list.get(0);
46      assertEquals ("FISH",cat.getCategoryId());
47      assertEquals ("Fish",cat.getName());
48      assertNotNull ("Expected product list.", cat.getProductList());
49      assertEquals (4, cat.getProductList().size());
50  
51      Product product = (Product)cat.getProductList().get(0);
52      assertEquals (2, product.getItemList().size());
53    }

55    /**
56     * This is a test case for iBATIS JIRA-250 "SELECT statement
57     * returns unexpected result when 'groupBy' and 'nullValue'
58     * are specified in resultMaps."
59     *
60     * The problem was that when a child object in a resultmap only
61     * contained null values it would still be created when one of the
62     * properties of the child object contained a nullValue attribute
63     * in the ResultMap. The nullValue would be applied before checking
64     * whether all properties of the child were 'null', so at least 1
65     * property would always be non-null (the reason for the
66     * nullValue attribute).
67     *
68     * The fix is to first check whether all properties of the child are
69     * 'null', and if the child object contains at least 1 non-null property
70     * to then only create the child object and apply the nullValue attribute.
71     *
72     * @throws Exception none should be thrown.
73     */
74    public void testGroupByJIRA250() throws Exception {
75      List list = sqlMap.queryForList("getAllProductCategoriesJIRA250",null);
76      Category cat = (Category)list.get(0);
77      assertEquals (0, cat.getProductList().size());
78    }
